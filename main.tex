% ---------- Header helper commands (used in main.tex only) ----------
\newcommand{\cvname}[1]{{\Huge\bfseries #1}}
\newcommand{\cvtitle}[1]{{\Large\color{cvgray} #1}}

% ---------- Reusable command for a job entry heading ----------
% \cvjob{Dates}{Duration}{Role}{Company}
\newcommand{\cvjob}[4]{%
  \noindent\textbf{#1} \; (#2) \; --- \; \textbf{#3} \; at \; \textbf{#4}\par\vspace{2pt}
}

% ---------- Company/school logo helper ----------
% \cvlogo[height]{path/to/file} — small inline image, right-aligned at
% the end of a heading line. ALWAYS keep the company name as plain text
% in the heading too; the logo is decoration, never a substitute.
% Default height (12.6pt) is tuned for use inside a \subsection* heading
% (set in \large). Logos placed in normal-size text (e.g. education.tex)
% or logos that read small at the default size need an explicit height
% override — pass it as the optional first argument, e.g. \cvlogo[18pt]{...}.
\newcommand{\cvlogo}[2][12.6pt]{%
  \hfill\raisebox{-0.25\height}{\includegraphics[height=#1]{#2}}%
}
